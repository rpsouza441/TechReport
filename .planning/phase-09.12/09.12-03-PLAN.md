  final List<Rat> _rats = [];
  List<Rat> get rats => List.unmodifiable(_rats);

    final index = _rats.indexWhere((r) => r.id == rat.id);

      _rats[index] = rat;
    } else {
      _rats.add(rat);
    }
  }

}

import 'package:flutter_test/flutter_test.dart';
import 'package:techreport/features/rat/domain/entities/rat.dart';
import 'package:techreport/features/rat/domain/services/rat_sync_coordinator.dart';
import 'package:techreport/features/sync/data/usecases/enqueue_rat_sync.dart';
import 'package:techreport/features/sync/domain/usecases/process_sync_queue.dart';

class MockSyncQueueRepository {
  final List<Map<String, dynamic>> queue = [];
  bool shouldFail = false;

  Future<void> enqueue({
    required String empresaId,
    required String usuarioId,
    required Map<String, dynamic> payload,
    required String operationType,
  }) async {
    if (shouldFail) throw Exception('Network error');
    queue.add({
      'empresa_id': empresaId,
      'usuario_id': usuarioId,
      'payload': payload,
      'operation_type': operationType,
      'status': 'pending',
      'created_at': DateTime.now().toIso8601String(),
    });
  }

  Future<List<Map<String, dynamic>>> getPending() async {
    return queue.where((item) => item['status'] == 'pending').toList();
  }

  Future<void> markProcessing(String id) async {
    final item = queue.firstWhere((q) => q['id'] == id);
    item['status'] = 'processing';
  }

  Future<void> markCompleted(String id) async {
    final item = queue.firstWhere((q) => q['id'] == id);
    item['status'] = 'completed';
  }

  Future<void> markFailed(String id, {String? error}) async {
    final item = queue.firstWhere((q) => q['id'] == id);
    item['status'] = 'failed';
    item['error'] = error;
    item['next_attempt_at'] = DateTime.now().add(const Duration(minutes: 5)).toIso8601String();
  }
}

void main() {
  late MockSyncQueueRepository syncQueueRepo;
  late RatSyncCoordinator syncCoordinator;

  setUp(() {
    syncQueueRepo = MockSyncQueueRepository();
    syncCoordinator = RatSyncCoordinator(
      enqueueRatSync: EnqueueRatSync(syncQueueRepository: syncQueueRepo),
      enqueueAssinaturaSync: EnqueueAssinaturaSync(syncQueueRepository: syncQueueRepo),
      processSyncQueue: ProcessSyncQueue(syncQueueRepository: syncQueueRepo),
    );
  });

  group('Sync queue enqueue', () {
    test('save enqueues RAT with pendingSync status', () async {
      final rat = _makeRat(id: 'rat-sync-1');

      await syncCoordinator.syncAfterSave(
        rat: rat,
        empresaId: 'emp-1',
        usuarioId: 'user-1',
      );

      expect(syncQueueRepo.queue.length, 1);
      expect(syncQueueRepo.queue.first['operation_type'], 'rat_save');
      expect(syncQueueRepo.queue.first['status'], 'pending');
    });

    test('delete enqueues RAT deletion', () async {
      final rat = _makeRat(id: 'rat-delete-1');

      await syncCoordinator.syncAfterDelete(
        rat: rat,
        empresaId: 'emp-1',
        usuarioId: 'user-1',
      );

      expect(syncQueueRepo.queue.length, 1);
      expect(syncQueueRepo.queue.first['operation_type'], 'rat_delete');
    });

    test('signature save enqueues signature sync', () async {
      final assinatura = _makeAssinatura(ratId: 'rat-sig-1');

      await syncCoordinator.syncAfterSignature(
        assinatura: assinatura,
        empresaId: 'emp-1',
        usuarioId: 'user-1',
        ratId: 'rat-sig-1',
      );

      expect(syncQueueRepo.queue.length, 1);
      expect(syncQueueRepo.queue.first['operation_type'], 'assinatura_save');
    });
  });

  group('Sync queue process', () {
    test('processSyncQueue processes pending items', () async {
      // Enqueue first
      final rat = _makeRat(id: 'rat-process-1');
      await syncCoordinator.syncAfterSave(
        rat: rat,
        empresaId: 'emp-1',
        usuarioId: 'user-1',
      );

      // Process queue
      final processResult = await ProcessSyncQueue(syncQueueRepository: syncQueueRepo).call(
        empresaId: 'emp-1',
        usuarioId: 'user-1',
        retryFailed: false,
      );

      expect(processResult.processedCount, 1);
    });

    test('retryFailed processes items with failed status', () async {
      // Enqueue and mark as failed
      final rat = _makeRat(id: 'rat-retry-1');
      await syncCoordinator.syncAfterSave(
        rat: rat,
        empresaId: 'emp-1',
        usuarioId: 'user-1',
      );
      syncQueueRepo.queue.first['status'] = 'failed';

      // Process with retry
      final processResult = await ProcessSyncQueue(syncQueueRepository: syncQueueRepo).call(
        empresaId: 'emp-1',
        usuarioId: 'user-1',
        retryFailed: true,
      );

      expect(processResult.processedCount, greaterThanOrEqualTo(1));
    });
  });

  group('Offline scenario', () {
    test('network failure does not lose RAT data', () async {
      syncQueueRepo.shouldFail = true;

      final rat = _makeRat(id: 'rat-offline-1');
      
      // Save should succeed locally even if sync fails
      // (RAT is saved to local DB before sync is attempted)
      expect(rat.syncStatus, RatSyncStatus.pendingSync);
      
      // After sync failure, RAT remains in local DB
      // Next app start should retry sync
    });

    test('multiple offline saves are queued correctly', () async {
      final rats = List.generate(5, (i) => _makeRat(id: 'rat-offline-$i'));
      
      for (final rat in rats) {
        await syncCoordinator.syncAfterSave(
          rat: rat,
          empresaId: 'emp-1',
          usuarioId: 'user-1',
        );
      }

      expect(syncQueueRepo.queue.length, 5);
      expect(
        syncQueueRepo.queue.where((q) => q['status'] == 'pending').length,
        5,
      );
    });
  });
}
